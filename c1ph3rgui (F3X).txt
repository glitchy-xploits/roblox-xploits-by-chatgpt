-- === SINGLE BUTTON CREATOR FUNCTION WITH AUTO-WRAP ===
local buttonWidth, buttonHeight = 120, 35
local spacingX, spacingY = 10, 10
local startX, startY = 15, 45
local currentX, currentY = startX, startY
local maxWidth = mainFrame.AbsoluteSize.X

function CreateButton(buttonName, onClick)
	-- Wrap to next row if too wide
	if (currentX + buttonWidth + spacingX) > maxWidth then
		currentX = startX
		currentY += buttonHeight + spacingY
	end

	-- Create button
	local button = Instance.new("TextButton")
	button.Size = UDim2.new(0, buttonWidth, 0, buttonHeight)
	button.Position = UDim2.new(0, currentX, 0, currentY)
	button.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
	button.TextColor3 = Color3.fromRGB(255, 255, 255)
	button.TextScaled = true
	button.Font = Enum.Font.SourceSansBold
	button.Text = buttonName
	button.Parent = mainFrame

	button.MouseButton1Click:Connect(onClick)

	-- Update position for next button
	currentX += buttonWidth + spacingX
end

-- Update wrapping dynamically when GUI resizes
mainFrame:GetPropertyChangedSignal("AbsoluteSize"):Connect(function()
	maxWidth = mainFrame.AbsoluteSize.X
end)

-- === MOBILE + PC DRAGGING SUPPORT ===
local UIS = game:GetService("UserInputService")
local dragging, dragInput, dragStart, startPos

local function updateDrag(input)
	if dragging then
		local delta = input.Position - dragStart
		mainFrame.Position = UDim2.new(
			startPos.X.Scale,
			startPos.X.Offset + delta.X,
			startPos.Y.Scale,
			startPos.Y.Offset + delta.Y
		)
	end
end

titleBackground.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 or
	   input.UserInputType == Enum.UserInputType.Touch then
		dragging = true
		dragStart = input.Position
		startPos = mainFrame.Position

		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

UIS.InputChanged:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseMovement or
	   input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end
end)

UIS.InputChanged:Connect(function(input)
	if input == dragInput then
		updateDrag(input)
	end
end)
